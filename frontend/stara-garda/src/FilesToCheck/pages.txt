import { useState } from "react";
import { useNavigate } from "react-router-dom";
import { createPlayer } from "../api/playerApi";
import CreatePlayerForm from "../components/players/CreatePlayerForm";

const CreatePlayer = () => {
    const navigate = useNavigate();
    const [submitting, setSubmitting] = useState(false);
    const [error, setError] = useState(null);

    const handleCreatePlayer = async (data) => {
        setError(null);
        setSubmitting(true);

        try {
            await createPlayer(data);
            navigate("/players");
        } catch (err) {
            const msg =
                err?.response?.data?.message ||
                "Nepoda≈ôilo se vytvo≈ôit hr√°ƒçe.";
            setError(msg);
        } finally {
            setSubmitting(false);
        }
    };

    return (
        <div className="container mt-4">
            <div className="row justify-content-center">
                <div className="col-md-6">

                    <h2 className="mb-4 text-center">
                        Vytvo≈ôit nov√©ho hr√°ƒçe
                    </h2>

                    {error && (
                        <div className="alert alert-danger">
                            {error}
                        </div>
                    )}

                    <CreatePlayerForm
                        onSubmit={handleCreatePlayer}
                        onCancel={() => navigate("/players")}
                        submitting={submitting}
                    />

                </div>
            </div>
        </div>
    );
};

export default CreatePlayer;

// src/pages/MatchDetailPage.jsx
import { useParams, useNavigate, useLocation } from "react-router-dom";
import { useState } from "react";

import { useMatchDetail } from "../hooks/useMatchDetail";
import { upsertMyRegistration } from "../api/MatchRegistrationApi";
import MatchDetail from "../components/matches/MatchDetail";
import { useNotification } from "../context/NotificationContext";
import { useCurrentPlayer } from "../hooks/useCurrentPlayer";
import { ExcuseReason, EXCUSE_REASON_OPTIONS } from "../constants/excuseReason";

const MatchDetailPage = () => {
    const { id } = useParams();
    const navigate = useNavigate();
    const location = useLocation();

    const { match, loading, error } = useMatchDetail(id);
    const [saving, setSaving] = useState(false);
    const [actionError, setActionError] = useState(null);

    const { showNotification } = useNotification();
    const { currentPlayer } = useCurrentPlayer();

    const playerName = currentPlayer?.fullName || "Hr√°ƒç";

    // jestli jde o uplynul√Ω z√°pas (p≈ôijde z router state)
    const isPast = location.state?.isPast === true;

    // üîπ sjednocen√≠ s backendem ‚Äì pojmenov√°n√≠ p≈ôesnƒõ jako DTO
    const playerMatchStatus = match?.playerMatchStatus ?? "NO_RESPONSE";
    const matchStatus = match?.matchStatus ?? null;      // nap≈ô. SCHEDULED / CANCELED / PLAYED

    // stav pro mod√°l (spoleƒçn√Ω pro omluvu i odhl√°≈°en√≠)
    const [showExcuseModal, setShowExcuseModal] = useState(false);
    const [selectedReason, setSelectedReason] = useState(ExcuseReason.JINE);
    const [excuseNote, setExcuseNote] = useState("");
    const [isUnregisterFlow, setIsUnregisterFlow] = useState(false); // true = ≈ôe≈°√≠me odhl√°≈°en√≠

    const handleRegister = async () => {
        if (!match) return;
        try {
            setSaving(true);
            setActionError(null);

            await upsertMyRegistration({
                matchId: match.id,
                team: null,
                excuseReason: null,
                excuseNote: null,
                unregister: false,
            });

            showNotification(
                `${playerName} byl √∫spƒõ≈°nƒõ p≈ôihl√°≈°en na z√°pas.`,
                "success"
            );
            navigate("/matches");
        } catch (err) {
            console.error(err);
            setActionError(
                err?.response?.data?.message ||
                "Nepoda≈ôilo se p≈ôihl√°sit na z√°pas."
            );
        } finally {
            setSaving(false);
        }
    };

    // spoleƒçn√° funkce pro otev≈ôen√≠ mod√°lu (omluva / odhl√°≈°en√≠)
    const openModal = (isUnregister) => {
        if (!match) return;

        setActionError(null);
        setIsUnregisterFlow(isUnregister);

        // p≈ôedvyplnit z backendu, pokud u≈æ existuje omluva
        const backendReason = match.excuseReason;
        const backendNote = match.excuseNote;

        setSelectedReason(
            backendReason && ExcuseReason[backendReason]
                ? backendReason
                : ExcuseReason.JINE
        );

        setExcuseNote(
            backendNote ??
            (isUnregister
                ? "Odhl√°≈°en√≠ ze z√°pasu."
                : "Nemohu se z√∫ƒçastnit.")
        );

        setShowExcuseModal(true);
    };

    // m√≠sto p≈ô√≠m√©ho odhl√°≈°en√≠ jen otev≈ôeme mod√°l v re≈æimu "unregister"
    const handleUnregister = () => {
        openModal(true);
    };

    // otev≈ôen√≠ mod√°lu pro klasickou omluvu
    const openExcuseModal = () => {
        openModal(false);
    };

    // odesl√°n√≠ z mod√°lu ‚Äì podle re≈æimu buƒè omluva, nebo odhl√°≈°en√≠ s d≈Øvodem
    const submitModalAction = async () => {
        if (!match) return;

        try {
            setSaving(true);
            setActionError(null);

            await upsertMyRegistration({
                matchId: match.id,
                team: null,
                excuseReason: selectedReason,
                excuseNote,
                unregister: isUnregisterFlow,
            });

            if (isUnregisterFlow) {
                showNotification(
                    `${playerName} byl odhl√°≈°en ze z√°pasu.`,
                    "info"
                );
            } else {
                showNotification(
                    `${playerName} se omluvil ze z√°pasu.`,
                    "warning"
                );
            }

            setShowExcuseModal(false);
            setIsUnregisterFlow(false);
            navigate("/matches");
        } catch (err) {
            console.error(err);
            setActionError(
                err?.response?.data?.message ||
                (isUnregisterFlow
                    ? "Nepoda≈ôilo se odhl√°sit ze z√°pasu."
                    : "Nepoda≈ôilo se omluvit ze z√°pasu.")
            );
        } finally {
            setSaving(false);
        }
    };

    const handleCloseModal = () => {
        setShowExcuseModal(false);
        setIsUnregisterFlow(false);
    };

    const modalTitle = isUnregisterFlow
        ? "Odhl√°sit se ze z√°pasu"
        : "Omluvit se ze z√°pasu";

    const submitButtonClass = isUnregisterFlow
        ? "btn btn-info"
        : "btn btn-warning";

    const submitButtonText = isUnregisterFlow
        ? "Odeslat odhl√°≈°en√≠"
        : "Odeslat omluvu";

    return (
        <>
            <MatchDetail
                match={match}
                playerMatchStatus={playerMatchStatus}  // üîπ jednoznaƒçn√Ω n√°zev
                matchStatus={matchStatus}             // üîπ stav z√°pasu
                loading={loading}
                error={error}
                actionError={actionError}
                onRegister={handleRegister}
                onUnregister={handleUnregister}
                onExcuse={openExcuseModal}
                saving={saving}
                isPast={isPast}
            />

            {showExcuseModal && (
                <div
                    className="modal fade show d-block"
                    tabIndex="-1"
                    role="dialog"
                >
                    <div className="modal-dialog" role="document">
                        <div className="modal-content">
                            <div className="modal-header">
                                <h5 className="modal-title">{modalTitle}</h5>
                                <button
                                    type="button"
                                    className="btn-close"
                                    onClick={handleCloseModal}
                                    disabled={saving}
                                />
                            </div>

                            <div className="modal-body">
                                <p>Vyberte d≈Øvod:</p>

                                {EXCUSE_REASON_OPTIONS.map((opt) => (
                                    <div className="form-check" key={opt.value}>
                                        <input
                                            className="form-check-input"
                                            type="radio"
                                            name="excuseReason"
                                            id={`excuse-${opt.value}`}
                                            value={opt.value}
                                            checked={selectedReason === opt.value}
                                            onChange={(e) =>
                                                setSelectedReason(e.target.value)
                                            }
                                            disabled={saving}
                                        />
                                        <label
                                            className="form-check-label"
                                            htmlFor={`excuse-${opt.value}`}
                                        >
                                            {opt.label}
                                        </label>
                                    </div>
                                ))}

                                <div className="mt-3">
                                    <label className="form-label">
                                        Pozn√°mka (voliteln√©):
                                    </label>
                                    <textarea
                                        className="form-control"
                                        rows={3}
                                        value={excuseNote}
                                        onChange={(e) =>
                                            setExcuseNote(e.target.value)
                                        }
                                        disabled={saving}
                                    />
                                </div>
                            </div>

                            <div className="modal-footer">
                                <button
                                    type="button"
                                    className="btn btn-secondary"
                                    onClick={handleCloseModal}
                                    disabled={saving}
                                >
                                    Zav≈ô√≠t
                                </button>
                                <button
                                    type="button"
                                    className={submitButtonClass}
                                    onClick={submitModalAction}
                                    disabled={saving}
                                >
                                    {submitButtonText}
                                </button>
                            </div>
                        </div>
                    </div>
                </div>
            )}
        </>
    );
};

export default MatchDetailPage;

import { useState } from "react";
import { useNavigate } from "react-router-dom";
import { registerUser } from "../api/authApi";

const RegisterPage = () => {
  const navigate = useNavigate();

  const [form, setForm] = useState({
    name: "",
    surname: "",
    email: "",
    password: "",
    passwordConfirm: ""
  });

  const [error, setError] = useState(null);
  const [loading, setLoading] = useState(false);
  const [success, setSuccess] = useState(false); // üëà NOV√ù STAV

  const handleChange = (e) => {
    setForm({
      ...form,
      [e.target.name]: e.target.value
    });
  };

  const handleSubmit = async (e) => {
    e.preventDefault();
    setError(null);
    setLoading(true);
    setSuccess(false);

    if (form.password !== form.passwordConfirm) {
      setError("Hesla se neshoduj√≠.");
      setLoading(false);
      return;
    }

    try {
      await registerUser(form);

      // M√çSTO NAVIGATE ZOBRAZ√çME HL√Å≈†KU
      setSuccess(true);

    } catch (err) {
      setError(err.message || "Registrace se nezda≈ôila.");
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="container mt-4">
      <div className="row justify-content-center">
        <div className="col-md-6 col-lg-5">
          <div className="card shadow p-4">
            <h3 className="text-center mb-4">Registrace</h3>

            {error && (
              <div className="alert alert-danger">
                {error}
              </div>
            )}

            {success && (
              <div className="alert alert-success">
                <strong>Registrace probƒõhla √∫spƒõ≈°nƒõ.</strong>
                <br />
                Byl V√°m zasl√°n e-mail s odkazem pro aktivaci √∫ƒçtu.
                Pros√≠m zkontrolujte svou schr√°nku (i slo≈æku Spam).

            {!success && (
              <form onSubmit={handleSubmit}>
                <div className="mb-3">
                  <label className="form-label">K≈ôestn√≠ jm√©no</label>
                  <input
                    type="text"
                    className="form-control"
                    name="name"
                    value={form.name}
                    onChange={handleChange}
                    required
                  />
                </div>

                <div className="mb-3">
                  <label className="form-label">P≈ô√≠jmen√≠</label>
                  <input
                    type="text"
                    className="form-control"
                    name="surname"
                    value={form.surname}
                    onChange={handleChange}
                    required
                  />
                </div>

                    name="email"

                    required
                  />
                </div>

                <div className="mb-3">
